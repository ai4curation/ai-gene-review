# Example batch gene input file
# Format: organism,gene[,uniprot_id][,alias]
#
# Simple format with just organism and gene symbol:
human,TP53
human,BRCA1

# With UniProt ID (for precise resolution):
human,ACTB,P60709

# For non-model organisms, use taxonomy code and UniProt ID with alias:
METEA,C5B1I4,,mllA
METEA,C5B1I5,,mllB
9BACT,F0JBF1,,HgcB

# Arabidopsis genes (using ARATH taxonomy code):
ARATH,BRI1
ARATH,FT

# Mouse genes:
mouse,Trp53
mouse,Brca1
